== Install cmake ===
cd /opt
wget https://github.com/Kitware/CMake/releases/download/v3.23.0/cmake-3.23.0.tar.gz
tar -xvzf cmake-3.23.0.tar.gz
cd cmake-3.23.0
make
make install
ln -s /opt/cmake-3.23.0/bin/* /usr/bin
